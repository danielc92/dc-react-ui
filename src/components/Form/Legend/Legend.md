### Features

- Uses native html tags for accessibility

### Default

```jsx
import React from 'react';
import Form from 'robust-react-ui/build/components/Form';
<Form>
  <Form.FieldSet>
    <Form.Legend>Form Legend</Form.Legend>
    <Form.Label>Input Label</Form.Label>
    <Form.Input />
  </Form.FieldSet>
</Form>;
```
