### Features

- supports multiple sizes

### Small

```jsx
import Link from 'robust-react-ui/build/components/Link';
<Link size="small" href="https://github.com">
  Visit github
</Link>;
```

### Default

```jsx
import Link from 'robust-react-ui/build/components/Link';
<Link size="medium" href="https://github.com">
  Visit github
</Link>;
```

### Large

```jsx
import Link from 'robust-react-ui/build/components/Link';
<Link size="large" href="https://github.com">
  Visit github
</Link>;
```
